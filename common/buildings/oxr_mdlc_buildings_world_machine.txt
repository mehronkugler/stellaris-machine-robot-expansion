		building_ancient_control_center
		building_ancient_palace

	}

	prerequisites = {
		"tech_planetary_government"
	}

	show_tech_unlock_if = {
		has_origin = oxr_mdlc_origin_world_machine_awakened
	}

	planet_modifier = {
		job_oxr_mdlc_world_machine_hibernation_slot_add = 3
		planet_housing_add = 5
		planet_amenities_add = 10
	}
	upgrades = {
		oxr_mdlc_world_machines_building_auto_capital_upgraded
	}
}

			owner = {

				free_building_slots = 0
				owner = {
					is_ai = yes
					has_monthly_income = {
						resource = food
						value > 100
					}
				}
			}
		}
	}
	resources = {
		category = oxr_mdlc_category_world_machines_buildings
		cost = { minerals = @xvcv_mdlc_district_cost_2 }

		upkeep = { energy = 1 }
	}
	ai_resource_production = {
		food = 10
		trigger = { always = yes }
	}
	prerequisites = {
		"tech_hydroponics"
		"oxr_mdlc_tech_world_machines_unlocked"
	}

			fire_on_action = { on_action = on_world_machine_popless_building_constructed }

xvcv_mdlc_species_create_machine_type = {
	create_species = {
		adjective = this
		name = this
		namelist = prev
		plural = this
		class = "MACHINE"
		portrait = this
		traits = { trait = trait_machine_unit }
		traits = this
		homeworld = prev.capital_scope
		effect = {
			modify_species = {
				remove_trait = trait_mechanical
				ideal_planet_class = this
				change_scoped_species = no
			}
		}
	}
}

oxr_mdlc_country_hack_authority_to_machine = {
	create_species = {
		adjective = this
		name = this
		namelist = prev
		plural = this
		class = "MACHINE"
		portrait = this
		traits = {
			trait = trait_machine_unit
		}
		traits = this
		homeworld = prev.capital_scope
	}
	last_created_species = {
		save_global_event_target_as = xvcv_temp_machine_species
	}
	change_dominant_species = { species = event_target:xvcv_temp_machine_species }
	country_add_ethic = ethic_gestalt_consciousness
	change_government = {
		authority = auth_machine_intelligence
		civics = {}
		cooldown = no
		remove_invalid_civics = yes
	}
	reset_policy_cooldowns = yes
	# Put back the bio robot species
	if = {
		limit = {
			exists = event_target:xvcv_mdlc_bio_robot_species
		}
		change_dominant_species = {
			species = xvcv_mdlc_bio_robot_species
		}
	}

	this.species = {
		if = {
			limit = { has_trait = trait_mechanical }
			change_species_characteristics = {
				remove_trait = trait_mechanical
			}
		}
		if = {
			limit = { NOT = { has_trait = trait_machine_unit } }
			change_species_characteristics = {
				add_trait = trait_machine_unit
			}
		}
	}
}
